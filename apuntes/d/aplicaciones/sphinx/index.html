Instalación¶

pip3 install -U sphinx
pip3 install -U sphinx_rtd_theme

sphinx-quickstart \
  --sep \
  --project apuntes \
  --author "Jorge P" \
  -v 20.05.21 \
  --release pub \
  --language es \
  --epub \
  --no-batchfile \
  --extensions \
  sphinx.ext.autodoc,\
  sphinx.ext.todo,\
  sphinx.ext.ifconfig,\
  sphinx.ext.imgconverter,\
  sphinx.ext.graphviz,\
  sphinx.ext.imgmath,\
  sphinx.ext.githubpages \
  apuntes

cd _static
mkdir css
echo < EOF >> css/my_theme.css
@import url("theme.css");
/*
.wy-nav-content {
  max-width: 90%
}
*/
EOF
cd ..

make html
make latexpdf
